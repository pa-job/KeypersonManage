CREATE TABLE SYSTEMCONFIG(
	`SID` int  primary key  auto_increment comment '主键',
	`SNAME`  varchar(26) not null  comment '名称',
	`SCONTENT` text  comment '内容',
	`STIME`  varchar(32) not null default '' comment '时间',
	`REMARK1`  varchar(32) not null default '' comment '备注字段2',
	`REMARK2`  varchar(12)  not null   default '' comment '备注字段1'
)engine=InnoDB default charset=utf8 comment='系统配置表';
